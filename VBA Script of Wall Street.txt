Sub multi()For Each ws In WorksheetsDim i As LongDim totalvolume As DoubleDim SummaryRow As IntegerSummaryRow = 2totalvolume = 0openingp = ws.Cells(2, 3).Valueclosingp = 0yearlychange = 0percentchange = 0maxvolume = 0maxvolumeticker = ""maxincrease = 0maxticker = ""minincrease = 999999999minticker = ""ws.Cells(1, 9).Value = "Ticker"ws.Cells(1, 10).Value = "Yearly Change"ws.Cells(1, 11).Value = "Percent Change"ws.Cells(1, 12).Value = "Total Stock Volume"ws.Cells(2, 15).Value = "Greatest % Increase"ws.Cells(3, 15).Value = "Greates % Decrease"ws.Cells(4, 15).Value = "Greatest Total Volume"ws.Cells(1, 16).Value = "Ticker"ws.Cells(1, 17).Value = "Value"lastrow = ws.Cells(Rows.Count, 1).End(xlUp).RowFor i = 2 To lastrowtotalvolume = totalvolume + ws.Cells(i, 7).ValueIf ws.Cells(i, 1).Value <> ws.Cells(i + 1, 1).Value Thenws.Cells(SummaryRow, 9) = ws.Cells(i, 1).Valuews.Cells(SummaryRow, 12) = totalvolumeclosingp = ws.Cells(i, 6).Valueyearlychange = closingp - openingpws.Cells(SummaryRow, 10) = yearlychangeIf openingp > 0 Then    percentchange = yearlychange / openingpElse    percentchange = 0End Ifws.Cells(SummaryRow, 11) = percentchangeIf ws.Cells(SummaryRow, 10) >= 0 Thenws.Cells(SummaryRow, 10).Interior.ColorIndex = 4Elsews.Cells(SummaryRow, 10).Interior.ColorIndex = 3End Ifopeningp = ws.Cells(i + 1, 3).ValueIf totalvolume > maxvolume Then      maxvolume = totalvolume      maxvolumeticker = ws.Cells(i, 1).Value    End Ifws.Cells(4, 17).Value = maxvolumews.Cells(4, 16).Value = maxvolumetickerIf percentchange < minincrease Then      minincrease = percentchange      minticker = ws.Cells(i, 1).Value    End Ifws.Cells(3, 17).Value = minincreasews.Cells(3, 16).Value = mintickerIf percentchange > maxincrease Then      maxincrease = percentchange      maxticker = ws.Cells(i, 1).Value    End Ifws.Cells(2, 17).Value = maxincreasews.Cells(2, 16).Value = maxtickerSummaryRow = SummaryRow + 1totalvolume = 0End IfNext iNext wsEnd Sub
